function loadCharacter() { const loader = new THREE.GLTFLoader(); loader.load( '/model.glb', // ← path function (gltf) { // ← success callback character = gltf.scene; // Enable shadows character.traverse((node) => { if (node.isMesh) { node.castShadow = true; node.receiveShadow = true; } }); scene.add(character); // Position & scale so it’s visible character.position.set(0, 0, 0); character.scale.set(1, 1, 1); // Hide loading text document.getElementById('loading').style.display = 'none'; console.log('Character loaded successfully!', gltf); }, function (xhr) { // ← progress console.log((xhr.loaded / xhr.total * 100) + '% loaded'); }, function (error) { // ← error console.error('Error loading model:', error); document.getElementById('loading').textContent = 'Error loading character. Check model.glb path.'; } ); }
